  const firebaseConfig = {
    apiKey: "YOUR_API_KEY",
    authDomain: "YOUR_PROJECT.firebaseapp.com",
    projectId: "YOUR_PROJECT_ID",
    storageBucket: "YOUR_PROJECT.appspot.com",
    messagingSenderId: "XXXX",
    appId: "XXXX"
  };

  const app = initializeApp(firebaseConfig);
  const db = getFirestore(app);

  let carsData = [];

  async function loadCars() {
    const querySnapshot = await getDocs(collection(db, "cars"));
    carsData = [];

    querySnapshot.forEach((doc) => {
      carsData.push(doc.data());
    });

    renderCars(carsData);

function previewImages(){
  const preview = document.getElementById("imagePreview");
  preview.innerHTML = "";

  const files = document.getElementById("carImages").files;

  Array.from(files).forEach(file => {
    const reader = new FileReader();

    reader.onload = function(e){
      const img = document.createElement("img");
      img.src = e.target.result;
      img.style.width = "120px";
      img.style.height = "80px";
      img.style.objectFit = "cover";
      img.style.borderRadius = "8px";
      img.style.border = "2px solid yellow";
      preview.appendChild(img);
    }

    reader.readAsDataURL(file);
  });
}

function contactDealer(){
  const name = document.getElementById("customerName").value;
  const email = document.getElementById("customerEmail").value;
  const car = document.getElementById("carSelect").value;

  if(!name || !email){
    alert("Please fill in your details.");
    return;
  }

  const subject = encodeURIComponent("Vehicle Inquiry - " + car);
  const body = encodeURIComponent(

    `My name is ${name}. I am interested in the ${car}.%0D%0A` +
    `Please contact me at: ${email}%0D%0A%0D%0AThank you.`
  );

  window.location.href = `mailto:Robertjeff521@gmail.com?subject=${subject}&body=${body}`;
}
  }

  function renderCars(cars) {
    const list = document.getElementById("carList");
    list.innerHTML = "";

    cars.forEach(car => {
      list.innerHTML += `
        <div class="car-card" data-type="${car.type}">
          ${car.image ? `<img src="${car.image}" style="width:100%; height:160px; object-fit:cover; border-radius:8px; margin-bottom:10px;">` : ''}
          <h3>${car.name}</h3>
          <p class="tag">${car.type} • ${car.year}</p>
          <p>$${car.price}</p>
        </div>
      `;
    });
  }

  function loginAdmin(){
  const user = document.getElementById("adminUser").value;

function addVehicle(){
  const name = document.getElementById("newCarName").value;
  const type = document.getElementById("newCarType").value;
  const year = document.getElementById("newCarYear").value;
  const price = document.getElementById("newCarPrice").value;
  const imageFile = document.getElementById("newCarImage").files[0];

  if(!name || !type || !year || !price){
    alert("Please fill all vehicle fields.");
    return;
  }

  const reader = new FileReader();

  reader.onload = function(e){
    carsData.push({
      name:name,
      type:type,
      year:year,
      price:price,
      image:e.target.result
    });

    renderCars(carsData);
    alert("Vehicle added successfully");
  }

  if(imageFile){
    reader.readAsDataURL(imageFile);
  } else {
    carsData.push({name,type,year,price});
    renderCars(carsData);
  }
}

window.filterCars = function () {
    let search = document.getElementById("search").value.toLowerCase();
    let filter = document.getElementById("filter").value;

    let filtered = carsData.filter(car => {
      let matchSearch = car.name.toLowerCase().includes(search);
      let matchFilter = filter === "all" || car.type === filter;
      return matchSearch && matchFilter;
    });

    renderCars(filtered);
  }

  loadCars();
